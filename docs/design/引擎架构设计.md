# 引擎架构设计文档

## 概述

对三国杀游戏引擎进行彻底重构。核心思想：**游戏是一个纯函数 `(GameState, Action) → { state, events }`**，状态完全可序列化，操作是数据，转换是纯函数。

## 设计目标

1. **可序列化** — GameState 可存储到 Redis，服务器重启后可从 `pending` 恢复
2. **原子操作粒度** — 所有状态修改走 Atom，Atom 可组合为技能，技能归属于角色/装备
3. **玩家级状态** — `PlayerState.vars` 存储技能运行时状态（发动次数、激活标记等）
4. **最优设计** — 不受现有代码约束，追求架构最优

## 设计约束

- 原子操作有可见性控制（摸牌者看牌面，其他人看数量）
- 原子操作有时间限制（超时自动默认动作）
- 两层校验：前端 src/shared/validation.ts（即时反馈）+ 引擎 validate（权威）
- 客户端通过事件日志同步状态（Event Sourcing）

### 注册 API

```typescript
const ATOM_REGISTRY = new Map<string, AtomDefinition>();

function registerAtom<A>(def: AtomDefinition<A>): void {
  ATOM_REGISTRY.set(def.type, def);
}

function applyAtom(state: GameState, atom: Atom): GameState {
  return ATOM_REGISTRY.get(atom.type)!.apply(state, atom);
}

function atomToEvents(state: GameState, atom: Atom): AtomEventResult {
  return ATOM_REGISTRY.get(atom.type)!.toEvents(state, atom);
}
```

### 注册示例

```typescript
// damage — 公开
registerAtom({
  type: 'damage',
  apply(state, atom) {
    const target = resolve(atom.target, state);
    const amount = resolve(atom.amount, state);
    return updatePlayer(state, target, p => ({ health: p.health - amount }));
  },
  toEvents(state, atom) {
    const target = resolve(atom.target, state);
    const amount = resolve(atom.amount, state);
    const source = resolve(atom.source, state);
    const evt = { id: genId(), type: 'damage', target, amount, source };
    return [evt, new Map(), evt] as const;  // 公开
  },
});

// draw — obscured
registerAtom({
  type: 'draw',
  apply(state, atom) {
    const player = resolve(atom.player, state);
    const count = resolve(atom.count, state);
    const drawn = state.zones.deck.splice(-count);
    return {
      ...state,
      zones: { ...state.zones, deck: state.zones.deck.slice(0, -count) },
      players: { ...state.players, [player]: {
        ...state.players[player],
        hand: [...state.players[player].hand, ...drawn],
      }},
    };
  },
  toEvents(state, atom) {
    const player = resolve(atom.player, state);
    const count = resolve(atom.count, state);
    const cards = state.players[player].hand.slice(-count).map(id => state.cardMap[id]);
    const server = { id: genId(), type: 'cardsDrawn', player, cards };
    const owner  = { id: server.id, type: 'cardsDrawn', player, cards };
    const others = { id: server.id, type: 'cardsDrawn', player, count };
    return [server, new Map([[player, owner]]), others] as const;
  },
});

// setVar — 私有
registerAtom({
  type: 'setVar',
  apply(state, atom) {
    const player = resolve(atom.player, state);
    const value = resolve(atom.value, state);
    return updatePlayer(state, player, p => ({
      vars: { ...p.vars, [atom.key]: value },
    }));
  },
  toEvents(state, atom) {
    const player = resolve(atom.player, state);
    const server = { id: genId(), type: 'varChanged', player, key: atom.key, value: resolve(atom.value, state) };
    const owner  = { id: server.id, type: 'varChanged', player, key: atom.key, value: server.value };
    return [server, new Map([[player, owner]]), null] as const;
  },
});
```

## 三、表达式系统（Expr）

```typescript
type Expr<T> =
  | T                                                   // 字面量
  | { $: 'ctx'; path: string }                          // SkillContext 引用
  | { $: 'event'; path: string }                        // 触发事件字段引用
  | { $: 'var'; player: Expr<string>; key: string }     // 玩家 vars
  | { $: 'count'; source: Expr<string> }                // 集合计数
  | { $: 'distance'; from: Expr<string>; to: Expr<string> }
  | { $: 'cardProp'; card: Expr<string>; prop: 'suit' | 'color' | 'type' }
  | { $: 'cond'; check: Condition; then: Expr<T>; else: Expr<T> }
  | { $: 'add'; left: Expr<number>; right: Expr<number> }
  | { $: 'sub'; left: Expr<number>; right: Expr<number> }
  | { $: 'handSize'; player: Expr<string> }
  | { $: 'aliveCount' };

// Condition 是独立联合类型，不属于 Expr
type Condition =
  | { equals: [unknown, unknown] }
  | { notEquals: [unknown, unknown] }
  | { gte: [Expr<number>, Expr<number>] }
  | { lte: [Expr<number>, Expr<number>] }
  | { gt: [Expr<number>, Expr<number>] }
  | { lt: [Expr<number>, Expr<number>] }
  | { hasVar: { player: Expr<string>; key: string } }
  | { hasTag: { player: Expr<string>; tag: string } }
  | { isAlive: Expr<string> }
  | { handEmpty: Expr<string> }
  | { hasValue: Expr<unknown> }
  | { and: Condition[] }
  | { or: Condition[] }
  | { not: Condition };

function resolve<T>(expr: Expr<T>, ctx: SkillContext, state: GameState): T;
```

字面量不需要 resolve，直接返回。对象形式才需要求值。

### 注册示例

```typescript
registerPhase({
  type: 'atoms',
  execute(state, phase, ctx) {
    let s = state;
    for (const atom of phase.ops) {
      s = applyAtom(s, atom);
    }
    return { state: s, events: [] };  // events 由 broadcast 层从 atomToEvents 收集
  },
});

registerPhase({
  type: 'prompt',
  execute(state, phase, ctx, plan, index) {
    return {
      state: {
        ...state,
        pending: {
          type: 'skillPrompt',
          skillId: ctx.skillId,
          player: ctx.self,
          execution: { phaseIndex: index, ctx },
          prompt: phase,
          timeout: phase.timeout ?? 15000,
          deadline: Date.now() + (phase.timeout ?? 15000),
          onTimeout: { type: 'skillChoice', player: ctx.self, choice: phase.defaultChoice ?? false },
        },
      },
      events: [],
    };
  },
});
```

### 触发流程

```typescript
function emitEvent(state: GameState, event: GameEvent): EngineResult {
  const triggers = state.triggers
    .filter(t => t.event === event.type && matchesFilter(state, event, t.filter))
    .sort((a, b) => b.priority - a.priority);

  let result: EngineResult = { state, events: [] };

  for (const trigger of triggers) {
    const skill = SKILL_REGISTRY.get(trigger.skillId)!;
    const ctx = buildSkillContext(state, event, trigger);
    const phases = skill.handler(ctx, result.state);

    result = executePlan(result.state, phases, ctx);
    if (result.state.pending) break;  // 暂停则停止
  }

  return result;
}
```

function engine(state: GameState, action: GameAction): EngineResult {
  // 有 pending → 处理 pending
  if (state.pending) return handlePending(state, action);

  // 无 pending → 处理新 action
  const err = validate(state, action);
  if (err) return { state, events: [], error: err };

  switch (action.type) {
    case 'playCard':    return handlePlayCard(state, action);
    case 'useSkill':    return handleUseSkill(state, action);
    case 'endTurn':     return handleEndTurn(state, action);
    case 'discard':     return handleDiscard(state, action);
  }
}

## 八、事件广播 ⚠️ broadcast() 独立函数已整合进 applyAtoms（ADR 0012），不再单独调用。设计意图不变。

### 事件生成

每个 Atom 执行后产生 `AtomEventResult`：
- `[0]` 服务端完整事件 → 写入 `serverLog`
- `[1]` 特殊视角 Map → 特定玩家看到的内容
- `[2]` 默认事件 → 不在 Map 中的玩家看到的内容（null = 不可见）

### 广播函数

```typescript
function broadcast(state: GameState, atoms: Atom[]): {
  state: GameState;
  playerEvents: Map<string, PlayerEvent[]>;
} {
  const playerEvents = new Map<string, PlayerEvent[]>();

  for (const atom of atoms) {
    const [serverEvent, playerMap, defaultEvent] = atomToEvents(state, atom);
    state = { ...state, serverLog: [...state.serverLog, serverEvent] };

    for (const player of state.playerOrder) {
      if (!playerEvents.has(player)) playerEvents.set(player, []);
      const specific = playerMap.get(player);
      if (specific) playerEvents.get(player)!.push(specific);
      else if (defaultEvent) playerEvents.get(player)!.push(defaultEvent);
    }

    state = applyAtom(state, atom);
  }

  return { state, playerEvents };
}
```
